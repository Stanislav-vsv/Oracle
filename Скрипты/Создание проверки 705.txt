3. Если проверка может выдать большое колличество строк, которое может переполнить таблицу DQ_MAIN, то надо
   добавить условие на превышение колличества строк. Если кол-во больше 1000 транзакция откатывается назад и 
   вставляется строка, с сообщением о переполнении. Пример проверка 705

 IF SQL%ROWCOUNT > 1000
   THEN
      INSERT INTO monitoring.DQ_MAIN (rev_id,
                                      list_id,
                                      rev_date,
                                      rev_datetime,
                                      F_STR1,
                                      F_NUM1,
                                      F_NUM2,
                                      F_NUM3,
                                      F_NUM4,
                                      F_STR2,
                                      error_flag)
         SELECT 2 AS rev_id,
                705 AS list_id,
                TO_NUMBER (TO_CHAR (SYSDATE, 'yyyymmdd')) rev_date,
                TO_CHAR (SYSDATE, 'dd.mm.yyyy hh24:mi:ss') rev_datetime,
                'проверка остановлена в связи с привышением количества строк' F_STR1,
                NULL,
                NULL,
                NULL,
                NULL,
                '<A href=''http://albertw:8080/dq/dit_revision.php?list_id=705''>Детали>></A>'
                   AS F_STR2,
                1 error_flag
           FROM DUAL; 
     END IF;

 IF SQL%ROWCOUNT > 1000  
   THEN
      ROLLBACK;
     END IF;

         COMMIT;
      END;

FROM dq_list, dq_main where dq_main.list_id = dq_list.list_id AND dq_list.list_id = 705

order by to_date(rev_date, 'yyyymmdd') desc;
